with Ada.Text_IO;

procedure  b3_3_3_1_uyg_4 is

   F : Float := 0.0;

   Alt_Eşik , Üst_Eşik , Ortalama , Bağıl_Hata : Float := 0.0;

   package F_In_Out is new Ada.Text_IO.Float_IO(num => Float);

   package Text_IO renames Ada.Text_IO;

begin

   Text_IO.Put(Item => "Float Değeri Giriniz : ");

   F_in_Out.Get(Item => F);

   Text_IO.Put(Item => "Girilen Float Değeri : ");

   F_In_Out.Put(Item => F , Fore => 1 , Aft => 8 , Exp =>3);

   Text_IO.New_Line;

   Text_IO.Put(Item => "Önceki Makine Değeri : ");

   F_In_Out.Put(Item => Float'Pred(F) , Fore => 1 , Aft => 8 , Exp =>3);

   Text_IO.New_Line;

   Text_IO.Put(Item => "Alt Eşik : ");

   Alt_Eşik := F - Float'Pred(F);

   F_In_Out.Put(Item => Alt_Eşik , Fore => 1 , Aft => 8 , Exp =>2);

   Text_IO.New_Line;

   Text_IO.Put(Item => "Sonraki Makine Değeri : ");

   F_In_Out.Put(Item => Float'Succ(F) , Fore => 1 , Aft => 8 , Exp =>3);

   Text_IO.New_Line;

   Text_IO.Put(Item => "Üst Eşik : ");

   Üst_Eşik := Float'Succ(F) - F;

   F_In_Out.Put(Item => Üst_Eşik , Fore => 1 , Aft => 8 , Exp =>2);

   Text_IO.New_Line;

   Text_IO.Put(Item => "Ortalama Fark : ");

   Ortalama := abs(Alt_Eşik + Üst_Eşik) /2.0;

   F_In_Out.Put(Item => Ortalama , Fore => 1 , Aft => 8 , Exp =>2);

   Text_IO.New_Line;

   Text_IO.Put(Item => "Ortalama Gerçekleşen % Duyarlık (%Bağıl Hata) : ");

   Bağıl_Hata := Ortalama * 100.0 /F;

   F_In_Out.Put(Item => Bağıl_Hata , Fore => 1 , Aft => 1 , Exp =>2);

   Text_IO.New_Line;

end b3_3_3_1_uyg_4;
