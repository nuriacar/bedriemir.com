diziYaz(tk , ' | ' , 'b8.2.10-uyg-2-sonuç-1');

yedek= tk.slice(0);// tk Dizisi Yedekleniyor !

tk=tk.slice(-18, -12);// Rekürsif Kopyalama ! (20-18, 20-12) = (2, 8) orijinal tk dizisi kayboldu, ama yedeği var !

diziYaz(tk , ' | ' , 'b8.2.10-uyg-2-sonuç-2');

tk = yedek.slice(0);// Orijinal tk Dizisinin Yeniden Oluşturulması !

tk = tk.slice(2, 8);// Normaiize Koordinatların Sınanması !


diziYaz(tk , ' | ' , 'b8.2.10-uyg-2-sonuç-3');

}

sayfaYüklendiktenSonraÇalıştır(diziYenile);
