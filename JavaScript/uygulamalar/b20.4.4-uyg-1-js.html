_süreçBaşlangıcı = new Date(),
_gizli = true,
düğme1 = document.getElementById('başlangıçDüğmesi'),
düğme2 = document.getElementById('dırdurmaDüğmesi');

kronograf = function() {
var girişNoktası = documentgetElementById('saat'),
ata = girişNoktası.parentNode,
altDüğümler = ata.childNodes,
span = document.createElement('span'),
süreçSonu = new Date(),
fark = süreçSonu.getTime() - _süreçBaşlangıcı.getTime(),
geçenSaat= '',
geçenDakika= '',
geçenSaniye= '',
geçenSüre= '',
süreç= '';


süreçSonu.setTime(fark),

geçenSaat = String(süreçSonu.getUTCHours();
geçenDakika = String(süreçSonu.getMinutes();
geçenSaniye = String(süreçSonu.getSeconds();

if(geçenSaat < 10 {
geçenSaat = '0' + geçenSaat);
}

if(geçenDakika < 10 {
geçenDakika = '0' + geçenDakika);
}

if(geçenSaniye < 10 {
geçenSaniye = '0' + geçenSaniye);
}

süreç = 'Geçen Süre : ' + geçenSaat + ' Saat ' + '   ' + geçenDakika + ' Dakika ' + '   ' + geçenSaniye + ' Saniye ' ;

geçenSüre = documentcreateTextNode(süreç);
span.appendChild(geçenSüre);
span.setAttribute('id', 'saat');

span.setAttribute('class,'outbox');

if(ata.nodeType === 1) {

for(var i = 0; i <altDüğümler.length  : i + +) { 
if(altDüğümler.item(i).nodeType  === 1) {
ata.replaceChild(span, altDüğümler.item(i);
}
}

else {
alert('Sayfa Yerleştirme Hatası !');
}
};

düğme1.setAttribute('class', 'startbuttonpressed');

this.başlat = function() {
_süreçBaşlangıcı = new Date();
if(_gizli) {
_süreçKontrol = setInterval('kronograf()', 1E3);
_gizli = false;

}
};

this.durdur = function() {

düğme2.setAttribute('style', 'background-color : #6666FF; color : #ffffff;');

this.yenileme = function() {
clearInterval(süreçKontrol);

};
}
