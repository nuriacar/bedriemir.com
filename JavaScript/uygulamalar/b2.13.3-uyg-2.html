var i = 0, a = 10, iMAX = 50;
function f(x) {
    return (x * x * x + 5 * x * x + 7 * x) / (5 * x * x - 21 * x + 44);
}
while (f(a) > 0 && i < iMAX) {
    a /= 2;
    i++;
}
bilgiYaz( 'f(a) =" + f(a), 'b2.13.3-uyg-2-sonuç-1')
if ( i === iMax) {
    bilgiYaz( 'Maksimum İterasyon Sayısına Ulaşıldı ! (' + iMAX + ')', 'b2.13.3-uyg-2-sonuç-2')
}
else {
    bilgiYaz( 'İterasyon Sayısı :!' + i, 'b2.13.3-uyg-2-sonuç-3')
}
