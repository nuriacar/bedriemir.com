function insertAfter(yeniElement, hedefElement) {

var ata= hedefElement.parentNode;

if(ata.lastChild === hedefElement) {

ata.appendChild(yeniElement);
}

else {

ata.insertBefore(yeniElement,hedefElement.nextSibling);
}
}
