def primes(m):
d = []
  
for num in range(2, m): # 2 den m e kadar iterasyon
for i in range(2, num): # sayının çarpanlarının araştırılması
if num % i == 0: # ilk çarpan
# j = num / i # ikinci çarpan
# print('{0} eşittir {1} * {2}'.format(num, i, j))
break # bir sonraki sayıya geç
else: # döngünün else kısmı
d.append(num)
# print('\n', num, 'bir asal sayıdır\n')  
 return d


q = primes(20)
print(q)
