listeQ = [16, 34, 2, 56, 44, 23]

print('f =     ', f)
listeP = [3, 6]

def önüneEkle(listeA, listeB):
    listeA[:0] = listeB

def sonunaEkle(listeA, listeB):
    listeA[len(listeA):] = listeB

önüneEkle(listeQ, listeP)

print('listeQ =', listeQ)
print('f =     ', f)

sonunaEkle(listeQ, listeP)

print('listeQ =', listeQ)
print('f =     ', f)    

listeQ = listeQ + listeP

print('listeQ =', listeQ)
print('f =     ', f)

sonunaEkle(listeQ, listeP)

print('listeQ =', listeQ)
print('f =     ', f)
