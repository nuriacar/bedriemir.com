def arkadaşListesi(tarih = '2 Aralık 2014', parti = 'Mangal', *arkadaş):

    maxVeri = len(arkadaş);

    i = 0;

    while i  maxVeri:

        print('\n',
' Sayın \n ', arkadaş[i] , ', \n',
''' {0:25} den itibaren başlayacak olan
  {1:3} partimize, lütfen katılmanızı rica ederim.
  Saygılarımla,
     Bedri Doğan Emir '''.format(tarih, parti));

        i = i + 1;


def main():

    arkadaşListesi('2 Ocak 2015 günü, saat 15.00', 'Çay',
        'Tülin Özer', 'Güler Karbeyaz',
        'Fortune Behar', 'Andonis Karaoğlu');

if __name__ == '__main__':

    main();
